import { Meta, ColorPalette, ColorItem } from "@storybook/blocks";
import { QUEN_UI_LIGHT_COLORS, QUEN_UI_DARK_COLORS } from "../colors"

<Meta title="Документация/Глобальные стили/Цвета" />

# Цвета

### Базовая цветовая палитра

<ColorPalette>
  {Object.entries(QUEN_UI_LIGHT_COLORS).map(([colorGroup, group]) => {
   return Object.keys(group).map((color) => {
      return (
      <div key={color}>
        <ColorItem
          subtitle={color}
          title={colorGroup}
          colors={{
            Light: QUEN_UI_LIGHT_COLORS[colorGroup][color],
            Dark: QUEN_UI_DARK_COLORS[colorGroup][color]
          }}
        />
      </div>
      )}
    )
  })}
</ColorPalette>
