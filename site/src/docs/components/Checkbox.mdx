#### Basic usage

Use Checkbox for binary selections, controlled via isChecked and onChange.

<Checkbox label="I agree to terms" />

```tsx
import { Checkbox } from "@quen-ui/components";

function Agreement() {
  const [checked, setChecked] = useState(false);

  return (
    <Checkbox
      checked={checked}
      onChange={setChecked}
      label="I agree to terms"
    />
  );
}
```

#### Base usage groups

Groups allow managing multiple related checkboxes with a single value state.

<BaseUsageGroup />

```tsx
import { Checkbox } from "@quen-ui/components";

const options = [
  { id: 1, name: "Option 1" },
  { id: 2, name: "Option 2" }
];

function Preferences() {
  const [selected, setSelected] = useState([1]);

  return (
    <Checkbox.Group
      options={options}
      value={selected}
      onChange={setSelected}
      getItemValue={(item) => item.id}
      getItemLabel={(item) => item.name}
      label="Select preferences"
    />
  );
}
```

#### Advanced Example with custom labels

Customize labels with icons, disable specific options, and render horizontally.

<AdvancedGroup />

```tsx
const complexOptions = [
  { code: "en", title: "English", icon: <Flag /> },

function LanguageSelect() {
  return (
    <Checkbox.Group
      options={complexOptions}
      direction="horizontal"
      getItemValue={(item) => item.code}
      getItemKey={(item) => item.code}
      getItemLabel={(item) => (
        <div className="language-option">
          {item.icon}
          <span>{item.title}</span>
        </div>
      )}
      getItemDisabled={(item) => item.disabled}
    />
  );
}
```

#### Combined "Select All" pattern

Demonstrates a common pattern where a parent checkbox controls a group.

<SelectAllPattern />

```tsx
function SelectAllPattern() {
  const [selected, setSelected] = useState<string[]>([]);
  const items = ["A", "B", "C"];

  const allSelected = selected.length === items.length;
  const someSelected = selected.length > 0 && !allSelected;

  return (
    <div>
      <Checkbox
        checked={allSelected}
        intermediate={someSelected}
        onChange={(checked) => {
          setSelected(checked ? [...items] : []);
        }}
        label="Select all"
      />

      <Checkbox.Group
        options={items}
        getItemLabel={(item) => item}
        getItemValue={(item) => item}
        value={selected}
        onChange={setSelected}
        direction="vertical"
      />
    </div>
  );
}
```

#### Best practices

- Use `Checkbox` for binary choices
- Use `Checkbox.Group` for multiple selections
- Always provide clear labels
- Group related options visually
- Use intermediate state only for partial selections
- Choose sizes based on layout density

#### API

```ts
interface ICheckboxGroupDefaultItem {
  key?: string | number;
  label: string;
  disabled?: boolean;
  value: string | number;
}

type TCheckboxGroupPropGetItemKey<TItem> = (
  item: TItem
) => string | number | undefined;

type TCheckboxGroupPropGetItemLabel<TItem> = (item: TItem) => React.ReactNode;
type TCheckboxGroupPropGetItemDisabled<TItem> = (
  item: TItem
) => boolean | undefined;
type TCheckboxGroupPropGetItemValue<
  TItem,
  VALUE extends string | number = string | number
> = (item: TItem) => VALUE;
```
