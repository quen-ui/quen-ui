import { Progress, Flex, Loader } from "@quen-ui/components";

#### Basic usage

```tsx
function Demo() {
  return <Progress value={65} color="green" />;
}
```

<Progress value={65} color="green" />

#### Size comparison

```tsx
<Flex gap="m" direction="column">
  <Progress size="xs" value={30} />
  <Progress size="s" value={50} />
  <Progress size="m" value={70} />
  <Progress size="l" value={100} />
</Flex>
```

#### With numeric value

Control dimensions using either design tokens or precise values:

```tsx
<Progress value={50} isShowInfo={true} size="l" color="red" />
```

<Progress value={50} isShowInfo={true} size="l" color="red" />

#### Custom label

```tsx
<Progress
  value={75}
  label={
    <div className="custom-label">
      <span>Processing...</span>
      <Loader size="xs" />
    </div>
  }
/>
```

#### Accessibility

- Uses `role="progressbar"`
- ARIA attributes:
```ts
  aria-valuenow={value}
  aria-valuemin={0}
  aria-valuemax={100}
```
