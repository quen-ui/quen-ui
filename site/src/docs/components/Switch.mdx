import { Loader, Flex, Switch } from "@quen-ui/components";

#### Basic usage

```tsx
function Demo() {
  const [notifications, setNotifications] = useState(true);
  return (
    <Switch
      value={notifications}
      onChange={(checked) => setNotifications(checked)}
      label="Email notifications"
      size="l"
    />
  );
}
```

<Switch defaultChecked label="Email notifications" size="l" />

#### Label positioning

```tsx
// Label before switch
<Switch label="Dark Mode" />

// Label after switch (default)
<Switch label="Dark Mode" labelPosition="after" />
```

#### Best practices

##### Keyboard navigation

- `Tab` focuses the switch
- `Space` toggles state
- `Enter` triggers onClick

##### Label guidelines

- Always include descriptive labels
- Use `labelPosition="before"` for right-to-left languages
- Combine with form field labels for complex forms
