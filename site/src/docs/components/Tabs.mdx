import { Tabs, Title, Badge } from "@quen-ui/components";
import { IconBell, IconMail } from "@tabler/icons-react";

#### Usage examples

Use `Tabs` to organize related content into multiple sections. Each `Tab` is linked with a `TabPanel` through the `value` prop.

<Tabs defaultValue="dashboard">
  <Tabs.TabsList className="p-remove-margin">
    <Tabs.Tab value="dashboard">Dashboard</Tabs.Tab>
    <Tabs.Tab value="analytics">Analytics</Tabs.Tab>
  </Tabs.TabsList>

<Tabs.TabPanel value="dashboard">
  <Title size="m">Dashboard</Title>
</Tabs.TabPanel>

  <Tabs.TabPanel value="analytics">
    <Title size="m">Analytics</Title>
  </Tabs.TabPanel>
</Tabs>

```tsx
<Tabs defaultValue="dashboard">

```

##### Styled tabs with icons and badges

Tabs can include icons or additional content (e.g. counters, badges). Use `leftContent` and `rightContent` props to decorate tab labels.

<Tabs.TabsList justify="space-between" className="p-remove-margin" >
  <Tabs.Tab
    value="notifications"
    leftContent={<IconBell />}
  >
    Alerts
  </Tabs.Tab>

  <Tabs.Tab
    value="messages"
    leftContent={<IconMail />}
    rightContent={<Badge>3</Badge>}
  >
    Messages
  </Tabs.Tab>
</Tabs.TabsList>

```tsx
<Tabs.TabsList justify="space-between">
  <Tabs.Tab value="notifications" leftContent={<IconBell />}>

    value="messages"
```

#### Keep mounted content

By default, hidden `TabPanels` are unmounted from the DOM. Use `keepMounted` to preserve their state when switching tabs.

<Tabs defaultValue="profile" keepMounted>
  <Tabs.TabsList>
    <Tabs.Tab value="profile">Profile</Tabs.Tab>
    <Tabs.Tab value="settings">Settings</Tabs.Tab>
  </Tabs.TabsList>

  <Tabs.TabPanel value="profile">
    Profile
  </Tabs.TabPanel>
  <Tabs.TabPanel value="settings">
    Settings
  </Tabs.TabPanel>
</Tabs>

```tsx
<Tabs defaultValue="profile" keepMounted>
  <Tabs.TabsList>
    <Tabs.Tab value="profile">Profile</Tabs.Tab>
    <Tabs.Tab value="settings">Settings</Tabs.Tab>
  </Tabs.TabsList>

  <Tabs.TabPanel value="profile">Profile</Tabs.TabPanel>
  <Tabs.TabPanel value="settings">Settings</Tabs.TabPanel>
</Tabs>
```

#### Best practices

##### 1. Tab naming conventions

  - Use short, descriptive labels (1–2 words)
  - Match the `value` prop exactly between `<Tab>` and `<TabPanel>`
  - Avoid using spaces or special characters in `value`

##### 2. Performance optimization

  - Prefer lazy loading heavy tab content to improve initial render time
  - Use `keepMounted` only when tab state must be preserved
```tsx
// Good - lazy loads heavy content
<Tabs.TabPanel value="reports">
  {activeTab === "reports" && <ReportGenerator />}
</Tabs.TabPanel>

// Better - use keepMounted strategically
<Tabs keepMounted={shouldPreserveState}>
```

##### 3. When to use

  - Organizing related content that is mutually exclusive
  - Switching between dashboards, reports, or content sections
  - Small to medium sets of options (2–6 tabs)

##### 4. When not to use

  - Too many tabs (>6) — use navigation or accordion instead
  - Sequential steps — use a Stepper component instead
  - Independent settings — use Checkboxes or Switches instead
