#### Signature

```ts
export const findBackgroundElement: (
  element: HTMLElement
) => { element: HTMLElement; color?: string; image?: string } | null;
```

#### Description

The function iterates through the chain of parent elements, starting with the specified `element`, and returns the first element that has:
  - opaque background (`background-colo`r ≠ `transparent` or `rgba(0,0,0,0)`)
  - background image (background-image ≠ none)
If none of the ancestors have a background style, the function returns null.

#### Usage Examples

```ts
const node = document.querySelector(".button");
const bg = findBackgroundElement(node);

if (bg) {
  console.log("Фон найден в элементе:", bg.element);
  if (bg.color) console.log("Цвет фона:", bg.color);
  if (bg.image) console.log("Фоновое изображение:", bg.image);
} else {
  console.log("Фон не найден");
}
```

##### Example of result

```json
{

  "color": "rgb(255, 255, 255)"

```

