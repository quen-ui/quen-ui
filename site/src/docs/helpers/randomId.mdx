#### Basic Example

```ts
import { randomId } from '@quen-ui/helpers';

const id = randomId();
// Example output: quen-ui-k3n9f2x7a
```

#### Custom Prefix

```ts
import { randomId } from '@quen-ui/helpers';

const buttonId = randomId("button-");
console.log(buttonId);
// Example output: button-r9p7x3z8c
```
