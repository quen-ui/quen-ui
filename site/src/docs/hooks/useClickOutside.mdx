#### Basic usage

Close a dropdown when clicking outside of its container.

```tsx
const Dropdown = () => {
  const ref = useRef<HTMLDivElement | null>(null);
  const [isOpen, setIsOpen] = useState(false);

  useOnClickOutside(ref, () => setIsOpen(false));

  return (
    <div ref={ref}>
      <Button onClick={() => setIsOpen(!isOpen)}>Toggle</Button>
      {isOpen && <div className="dropdown">Menu content</div>}
    </div>
  );
};
```

#### Multiple refs

You can pass an array of refs if you want to track multiple elements.

```tsx
const Modal = () => {
  const modalRef = useRef<HTMLDivElement | null>(null);
  const footerRef = useRef<HTMLDivElement | null>(null);

  useOnClickOutside([modalRef, footerRef], () => {
    console.log("Clicked outside modal and footer");
  });

  return (
    <>
      <div ref={modalRef}>Modal content</div>
      <div ref={footerRef}>Footer buttons</div>
    </>
  );
};
```

#### Ignoring scrollbar clicks

Prevent triggering the handler when a click happens on the scrollbar area.

```tsx
useOnClickOutside(ref, () => {
  console.log("Clicked outside");
}, { ignoreScrollbar: true });

```


#### Excluding specific elements

Skip triggering the handler when the click happens inside `excludeRef`.


```tsx
const ref = useRef<HTMLDivElement | null>(null);
const excludeRef = useRef<HTMLButtonElement | null>(null);

useOnClickOutside(ref, () => {
  console.log("Outside clicked, excluding button");
}, { excludeRef });

```

#### Conditional activation

Enable or disable the listener dynamically.

```tsx
useOnClickOutside(ref, () => {
  console.log("Only active when dropdown is open");
}, { isActive: isOpen });

```
