#### Basic usage

Ideal for optimizing performance in high-frequency events like search inputs or window resizes.

```tsx
import { useState } from "react";
import { useDebounce } from "@quen-ui/hooks";

function SearchExample() {
  const [query, setQuery] = useState("");

  const handleSearch = (searchQuery: string) => {
    console.log("Executing API call for:", searchQuery);
    // Perform heavy computation or API call here
  };

  // Returns a stable debounced function
  const debouncedSearch = useDebounce(handleSearch, 300);

  return (
    <input
      type="text"
      value={query}
      placeholder="Type to search..."
      onChange={(e) => {
        setQuery(e.target.value);
        debouncedSearch(e.target.value);
      }}
    />
  );
}
```

#### Advanced usage: Form Validation (Flush before submit)

When using debounce for async validation (e.g., checking email uniqueness), you can use the `flush()` method to ensure the validation completes before the form is submitted.

```tsx
import { useDebounce } from "@quen-ui/hooks";

function FormValidationExample() {
  const validateEmail = async (email: string) => {
    const isTaken = await api.checkEmail(email);
    if (isTaken) setError("Email is already taken");
  };

  // Debounce validation by 500ms
  const debouncedValidate = useDebounce(validateEmail, 500);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    // Force immediate execution of any pending validation
    // before allowing the form to submit
    debouncedValidate.flush();

    // Proceed with submission logic...
  };

  return (
    <form onSubmit={handleSubmit}>
      <input type="email" onChange={(e) => debouncedValidate(e.target.value)} />
      <button type="submit">Submit</button>
    </form>
  );
}
```

#### API

```ts
interface DebouncedFn<T extends (...args: any[]) => any> {
  (...args: Parameters<T>): void;
  /** Cancels any pending execution of the debounced function. */
  cancel: () => void;
  /** Immediately executes the pending callback, clearing the timer. */
  flush: () => void;
}

useDebounce<T extends (...args: any[]) => any>(
  callback: T,
  delay: number
): DebouncedFn<T>
```
